    function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data = JSON.parse(e.postData.contents);

    // first response: write headers
    if (sheet.getLastRow() === 0) {
      var headers = ['提交時間'];
      var matrixKeys = Object.keys(data.matrix).sort();
      matrixKeys.forEach(function(k) { headers.push(k); });
      var personalKeys = Object.keys(data.personal);
      personalKeys.forEach(function(k) { headers.push(k); });
      sheet.appendRow(headers);
    }

    // read existing headers to match column order
    var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    var row = [];
    headers.forEach(function(h) {
      if (h === '提交時間') {
        row.push(data.timestamp || new Date().toISOString());
      } else if (data.matrix && data.matrix[h] !== undefined) {
        row.push(data.matrix[h]);
      } else if (data.personal && data.personal[h] !== undefined) {
        row.push(data.personal[h]);
      } else {
        row.push('');
      }
    });

    sheet.appendRow(row);

    return ContentService
      .createTextOutput(JSON.stringify({status: 'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({status: 'error', message: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
